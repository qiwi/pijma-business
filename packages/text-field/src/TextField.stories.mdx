import { TextField as StatelessTextField } from './TextField'
import { Meta, ArgsTable, Canvas, Story } from '@storybook/addon-docs'
import { View } from '@pijma/react-native'
import { useState } from 'react'

<Meta
  title="Формы/Текстовое поле"
  component={StatelessTextField}
  decorators={[(Story) => <View width={['100%', 320]}>{Story()}</View>]}
/>

export const TextField = (props) => {
  const [value, setValue] = useState(props.value || '')
  return <StatelessTextField {...props} value={value} onModify={setValue} />
}
